# CARD SERVICE - COMPLETE LEARNING GUIDE
**Port: 5002 | Database: credvault_cards**

---

## What This Service Does

The Card Service manages **credit cards and their transactions**. It's where users add, view, update, and delete their credit cards.

**Capabilities:**
- Add credit cards (with validation, Luhn check, network detection)
- View cards (user's own cards)
- Update card details (nickname, default)
- Delete cards (soft delete with balance check)
- Add/view card transactions (purchases, payments, refunds)
- Admin: manage card issuers, update card limits

```csharp
public class CreditCard : BaseEntity  // BaseEntity provides Id, CreatedAtUtc, UpdatedAtUtc
{
    public Guid UserId { get; set; }              // FK to identity_users (cross-service)

    public string Last4 { get; set; }             // Last 4 digits
    
    [MaxLength(32)]

    public decimal CreditLimit { get; set; }      // Total credit limit
    public decimal OutstandingBalance { get; set; } // Current balance owed
    
    public int BillingCycleStartDay { get; set; } // When billing cycle starts (1-31)
    
    public bool IsDefault { get; set; }           // Primary card for user
    
    public bool IsVerified { get; set; }          // Card ownership verified
    public DateTime? VerifiedAtUtc { get; set; } // When verified

    public DateTime? DeletedAtUtc { get; set; }
```

**Important Patterns to Notice:**
1. **Soft Delete:** `IsDeleted` flag instead of hard delete
2. **Cross-service FK:** `UserId` references identity service's user
3. **Same-service FK:** `IssuerId` references CardIssuer (can have FK constraint)

#### CreateCardCommandHandler.cs (Full Breakdown)

```csharp
public class CreateCardCommandHandler : IRequestHandler<CreateCardCommand, ApiResponse<CardDto>>
{
    private readonly ICardRepository _cardRepository;
    private readonly IPublishEndpoint _publishEndpoint;
    private readonly ILogger<CreateCardCommandHandler> _logger;
    
    public CreateCardCommandHandler(
        ICardRepository cardRepository,
        IPublishEndpoint publishEndpoint,
        ILogger<CreateCardCommandHandler> logger)
    {
        _cardRepository = cardRepository;
        _publishEndpoint = publishEndpoint;
        _logger = logger;
    }
    
    public async Task<ApiResponse<CardDto>> Handle(
        CreateCardCommand request, 
        CancellationToken cancellationToken)
    {
        // ==========================================
        // STEP 1: Validate Expiration Date
        // ==========================================
        if (!CardHelpers.IsValidExpiration(request.ExpMonth, request.ExpYear))
        {
            return ApiResponse<CardDto>.Fail("Card has expired or invalid expiration date");
        }
        
        // ==========================================
        // STEP 2: Detect Card Network (Visa vs Mastercard)
        // ==========================================
        var detectedNetwork = CardHelpers.DetectCardNetwork(request.CardNumber);
        if (detectedNetwork == CardNetwork.Unknown)
        {
            return ApiResponse<CardDto>.Fail("Unable to detect card network");
        }
        
        // ==========================================
        // STEP 3: Validate with Luhn Algorithm
        // ==========================================
        if (!CardHelpers.IsValidCardNumber(request.CardNumber))
        {
            return ApiResponse<CardDto>.Fail("Invalid card number");
        }
        
        // ==========================================
        // STEP 4: Check for Duplicate Card
        // ==========================================
        var last4 = CardHelpers.GetLast4Digits(request.CardNumber);
        var isDuplicate = await _cardRepository.CardExistsForUserAsync(
            request.UserId, last4, request.ExpMonth, request.ExpYear);
        
        if (isDuplicate)
        {
            return ApiResponse<CardDto>.Fail("This card is already added");
        }
        
        // ==========================================
        // STEP 5: Verify Issuer Exists
        // ==========================================
        var issuer = await _cardRepository.GetIssuerByIdAsync(request.IssuerId);
        if (issuer == null)
        {
            return ApiResponse<CardDto>.Fail("Invalid card issuer");
        }
        
        // ==========================================
        // STEP 6: Handle Default Card Logic
        // ==========================================
        if (request.IsDefault)
        {
            // Unset any existing default card
            var existingDefault = await _cardRepository.GetDefaultCardAsync(request.UserId);
            if (existingDefault != null)
            {
                existingDefault.IsDefault = false;
                await _cardRepository.UpdateAsync(existingDefault);
            }
        }
        
        // ==========================================
        // STEP 7: Create Card Entity
        // ==========================================
        var card = new CreditCard
        {
            Id = Guid.NewGuid(),
            UserId = request.UserId,
            IssuerId = request.IssuerId,
            CardholderName = request.CardholderName,
            ExpMonth = request.ExpMonth,
            ExpYear = request.ExpYear,
            Last4 = last4,
            MaskedNumber = CardHelpers.MaskCardNumber(request.CardNumber),
            CreditLimit = 0,  // Set later by admin
            OutstandingBalance = 0,
            BillingCycleStartDay = request.BillingCycleStartDay,
            IsDefault = request.IsDefault,
            IsVerified = false,  // Requires verification
            IsDeleted = false,
            CreatedAtUtc = DateTime.UtcNow,
            UpdatedAtUtc = DateTime.UtcNow
        };
        
        // ==========================================
        // STEP 8: Save to Database
        // ==========================================
        await _cardRepository.AddAsync(card);
        
        // ==========================================
        // STEP 9: Publish Event
        // ==========================================
        await _publishEndpoint.Publish(new ICardAdded
        {
            CorrelationId = Guid.NewGuid(),
            CardId = card.Id,
            UserId = card.UserId,
            Last4 = card.Last4,
            CardholderName = card.CardholderName,
            Network = detectedNetwork,
            Timestamp = DateTime.UtcNow
        });
        
        _logger.LogInformation("Card added: {CardId} for user {UserId}", card.Id, card.UserId);
        
        // ==========================================
        // STEP 10: Return Response
        // ==========================================
        return ApiResponse<CardDto>.Success(
            CardMapping.ToDto(card, issuer.Name, detectedNetwork),
            "Card added successfully"
        );
    }
}
```

### 3. CardHelpers.cs (Card Validation Logic)

**Location:** `CardService.Application/Helpers/CardHelpers.cs`

**This is the brain of card validation. Read it to understand:**

```csharp
public static class CardHelpers
{
    // ==========================================
    // Luhn Algorithm - Credit Card Validation
    // ==========================================
    public static bool IsValidCardNumber(string cardNumber)
    {
        // Remove spaces and dashes
        cardNumber = cardNumber.Replace(" ", "").Replace("-", "");
        
        // Must be numeric and 13-19 digits
        if (!cardNumber.All(char.IsDigit) || cardNumber.Length < 13 || cardNumber.Length > 19)
            return false;
        
        // Luhn algorithm
        int sum = 0;
        bool alternate = false;
        
        for (int i = cardNumber.Length - 1; i >= 0; i--)
        {
            int digit = int.Parse(cardNumber[i].ToString());
            
            if (alternate)
            {
                digit *= 2;
                if (digit > 9)
                    digit -= 9;
            }
            
            sum += digit;
            alternate = !alternate;
        }
        
        return sum % 10 == 0;
    }
    
    // ==========================================
    // Network Detection (Visa vs Mastercard)
    // ==========================================
    public static CardNetwork DetectCardNetwork(string cardNumber)
    {
        cardNumber = cardNumber.Replace(" ", "").Replace("-", "");
        
        // Visa: Starts with 4
        if (cardNumber.StartsWith("4"))
            return CardNetwork.Visa;
        
        // Mastercard: Starts with 51-55 or 2221-2720
        if (cardNumber.Length >= 2)
        {
            int firstTwo = int.Parse(cardNumber.Substring(0, 2));
            if (firstTwo >= 51 && firstTwo <= 55)
                return CardNetwork.Mastercard;
        }
        
        if (cardNumber.Length >= 4)
        {
            int firstFour = int.Parse(cardNumber.Substring(0, 4));
            if (firstFour >= 2221 && firstFour <= 2720)
                return CardNetwork.Mastercard;
        }
        
        return CardNetwork.Unknown;
    }
    
    // ==========================================
    // Extract Last 4 Digits
    // ==========================================
    public static string GetLast4Digits(string cardNumber)
    {
        cardNumber = cardNumber.Replace(" ", "").Replace("-", "");
        return cardNumber[^4..];  // Last 4 characters
    }
    
    // ==========================================
    // Mask Card Number
    // ==========================================
    public static string MaskCardNumber(string cardNumber)
    {
        cardNumber = cardNumber.Replace(" ", "").Replace("-", "");
        var last4 = cardNumber[^4..];
        return $"****-****-****-{last4}";
    }
    
    // ==========================================
    // Expiration Validation
    // ==========================================
    public static bool IsValidExpiration(int month, int year)
    {
        if (month < 1 || month > 12)
            return false;
        
        var now = DateTime.UtcNow;
        var cardExpiry = new DateTime(year, month, 1).AddMonths(1).AddDays(-1);
        
        // Card is valid through the last day of expiration month
        return cardExpiry >= now.Date;
    }
}
```

### 4. DeleteCardCommand & Handler (Soft Delete Pattern)

**Location:** `CardService.Application/Commands/DeleteCard/`

**Important:** This teaches soft delete with validation.

```csharp
public class DeleteCardCommandHandler : IRequestHandler<DeleteCardCommand, ApiResponse<bool>>
{
    private readonly ICardRepository _cardRepository;
    
    public async Task<ApiResponse<bool>> Handle(DeleteCardCommand request, CancellationToken cancellationToken)
    {
        var card = await _cardRepository.GetByIdAsync(request.CardId);
        
        if (card == null)
            return ApiResponse<bool>.Fail("Card not found");
        
        // Validate ownership
        if (card.UserId != request.UserId)
            return ApiResponse<bool>.Fail("Unauthorized");
        
        // Check for outstanding balance
        if (card.OutstandingBalance > 0)
            return ApiResponse<bool>.Fail("Cannot delete card with outstanding balance");
        
        // Check for recent transactions
        var transactions = await _cardRepository.GetTransactionsByCardIdAsync(card.Id, 1, 1);
        if (transactions.Any())
            return ApiResponse<bool>.Fail("Cannot delete card with transactions. Please contact support.");
        
        // Soft delete
        card.IsDeleted = true;
        card.DeletedAtUtc = DateTime.UtcNow;
        await _cardRepository.UpdateAsync(card);
        
        return ApiResponse<bool>.Success(true, "Card deleted successfully");
    }
}
```

### 5. AddCardTransactionCommand & Handler

**Location:** `CardService.Application/Commands/AddCardTransaction/`

```csharp
public class AddCardTransactionCommandHandler : IRequestHandler<AddCardTransactionCommand, ApiResponse<TransactionDto>>
{
    public async Task<ApiResponse<TransactionDto>> Handle(AddCardTransactionCommand request, CancellationToken cancellationToken)
    {
        var card = await _cardRepository.GetByIdAsync(request.CardId);
        
        if (card == null || card.IsDeleted)
            return ApiResponse<TransactionDto>.Fail("Card not found");
        
        // Create transaction
        var transaction = new CardTransaction
        {
            Id = Guid.NewGuid(),
            CardId = request.CardId,
            UserId = request.UserId,
            Type = request.Type,
            Amount = request.Amount,
            Description = request.Description,
            DateUtc = request.Date ?? DateTime.UtcNow
        };
        
        await _cardRepository.AddTransactionAsync(transaction);
        
        // Update card balance based on transaction type
        switch (request.Type)
        {
            case CardTransactionType.Purchase:
                card.OutstandingBalance += request.Amount;
                break;
                
            case CardTransactionType.Payment:
                card.OutstandingBalance -= request.Amount;
                // Don't go below zero
                if (card.OutstandingBalance < 0)
                    card.OutstandingBalance = 0;
                break;
                
            case CardTransactionType.Refund:
                card.OutstandingBalance -= request.Amount;
                if (card.OutstandingBalance < 0)
                    card.OutstandingBalance = 0;
                break;
        }
        
        await _cardRepository.UpdateAsync(card);
        
        return ApiResponse<TransactionDto>.Success(
            CardMapping.ToTransactionDto(transaction),
            "Transaction added"
        );
    }
}
```

### 1. CardDbContext.cs (With Soft Delete Filter)

**Location:** `CardService.Infrastructure/Persistence/Sql/CardDbContext.cs`

**Key Learning:** Global query filter for soft delete.

```csharp
public class CardDbContext : DbContext
{
    public DbSet<CreditCard> Cards { get; set; }
    public DbSet<CardIssuer> Issuers { get; set; }
    public DbSet<CardTransaction> Transactions { get; set; }
    
    protected override void OnModelCreating(ModelBuilder modelBuilder)
    {
        base.OnModelCreating(modelBuilder);
        
        // ==========================================
        // CreditCard Configuration
        // ==========================================
        modelBuilder.Entity<CreditCard>(entity =>
        {
            entity.ToTable("credit_cards");
            
            entity.HasKey(e => e.Id);
            
            // Index for user's cards (most common query)
            entity.HasIndex(e => e.UserId);
            
            // Composite index for default card lookup
            entity.HasIndex(e => new { e.UserId, e.IsDefault });
            
            // Enum conversions
            entity.Property(e => e.Network)
                .HasConversion<int>();
            
            // SOFT DELETE - Global Query Filter
            // This automatically filters out deleted cards unless explicitly included
            entity.HasQueryFilter(e => !e.IsDeleted);
        });
        
        // ==========================================
        // CardIssuer Configuration
        // ==========================================
        modelBuilder.Entity<CardIssuer>(entity =>
        {
            entity.ToTable("card_issuers");
            entity.HasKey(e => e.Id);
            
            entity.Property(e => e.Network)
                .HasConversion<int>();
        });
        
        // ==========================================
        // CardTransaction Configuration
        // ==========================================
        modelBuilder.Entity<CardTransaction>(entity =>
        {
            entity.ToTable("card_transactions");
            entity.HasKey(e => e.Id);
            
            // Indexes for common queries
            entity.HasIndex(e => e.CardId);
            entity.HasIndex(e => e.UserId);
            entity.HasIndex(e => e.DateUtc);
            
            entity.Property(e => e.Type)
                .HasConversion<int>();
        });
    }
}
```

**The Magic of Query Filters:**
```csharp
// Without filter - you manually check IsDeleted everywhere
var card = await _context.Cards.FirstOrDefaultAsync(c => c.Id == id && !c.IsDeleted);

// With global filter - automatically filtered
var card = await _context.Cards.FirstOrDefaultAsync(c => c.Id == id);
// Deleted cards are NEVER returned unless you explicitly ignore the filter

// To include deleted records (for admin views)
var card = await _context.Cards
    .IgnoreQueryFilters()
    .FirstOrDefaultAsync(c => c.Id == id);
```

### 2. SqlCardRepository.cs

**Location:** `CardService.Infrastructure/Persistence/Sql/Repositories/SqlCardRepository.cs`

**Key Implementation Details:**

```csharp
public class SqlCardRepository : ICardRepository
{
    private readonly CardDbContext _context;
    
    public SqlCardRepository(CardDbContext context)
    {
        _context = context;
    }
    
    public async Task<CreditCard?> GetByIdAsync(Guid id)
    {
        // Uses the global query filter - deleted cards excluded
        return await _context.Cards
            .AsNoTracking()
            .Include(c => c.Issuer)  // Load related issuer
            .FirstOrDefaultAsync(c => c.Id == id);
    }
    
    public async Task<List<CreditCard>> GetByUserIdAsync(Guid userId, bool includeDeleted = false)
    {
        var query = _context.Cards.AsNoTracking();
        
        // Bypass the global filter if needed
        if (includeDeleted)
            query = query.IgnoreQueryFilters();
        
        return await query
            .Include(c => c.Issuer)
            .Where(c => c.UserId == userId)
            .OrderByDescending(c => c.IsDefault)
            .ThenByDescending(c => c.CreatedAtUtc)
            .ToListAsync();
    }
    
    public async Task<bool> CardExistsForUserAsync(
        Guid userId, 
        string last4, 
        int expMonth, 
        int expYear)
    {
        return await _context.Cards
            .IgnoreQueryFilters()  // Check ALL cards including deleted
            .AnyAsync(c => 
                c.UserId == userId &&
                c.Last4 == last4 &&
                c.ExpMonth == expMonth &&
                c.ExpYear == expYear);
    }
}
```

```csharp
        var message = context.Message;
        
        // Get all cards for this user
        var cards = await _cardRepository.GetByUserIdAsync(message.UserId, includeDeleted: true);
        
        foreach (var card in cards)
        {
            card.IsDeleted = true;
            card.DeletedAtUtc = DateTime.UtcNow;
            await _cardRepository.UpdateAsync(card);
        }
        
        // Note: We don't actually delete, we soft delete
        // This preserves data for auditing/compliance
    }
}
```

### 3. CardDeductionSagaConsumer.cs

**Location:** `CardService.Infrastructure/Consumers/CardDeductionSagaConsumer.cs`

**What it does:** Part of payment saga - deducts amount from card balance.

```csharp
public class CardDeductionSagaConsumer : IConsumer<IStartCardDeduction>
{
    private readonly ICardRepository _cardRepository;
    private readonly IPublishEndpoint _publishEndpoint;
    
    public async Task Consume(ConsumeContext<IStartCardDeduction> context)
    {
        var message = context.Message;
        
        var card = await _cardRepository.GetByIdAsync(message.CardId);
        
        if (card == null || card.IsDeleted)
        {
            // Card not found - publish failure
            await _publishEndpoint.Publish(new ICardDeductionFailed
            {
                CorrelationId = message.CorrelationId,
                CardId = message.CardId,
                Error = "Card not found"
            });
            return;
        }
        
        // Check available credit
        var availableCredit = card.CreditLimit - card.OutstandingBalance;
        if (availableCredit < message.Amount)
        {
            // Insufficient credit
            await _publishEndpoint.Publish(new ICardDeductionFailed
            {
                CorrelationId = message.CorrelationId,
                CardId = message.CardId,
                Error = "Insufficient credit"
            });
            return;
        }
        
        // Deduct the amount
        card.OutstandingBalance += message.Amount;
        await _cardRepository.UpdateAsync(card);
        
        // Publish success
        await _publishEndpoint.Publish(new ICardDeducted
        {
            CorrelationId = message.CorrelationId,
            CardId = card.Id,
            AmountDeducted = message.Amount,
            NewBalance = card.OutstandingBalance,
            Timestamp = DateTime.UtcNow
        });
    }
}
```
