# NOTIFICATION SERVICE - COMPLETE LEARNING GUIDE
**Port: 5005 | Database: Per-service (shared)**

---

## What This Service Does

The Notification Service handles **emails, SMS, and audit logging**. It listens to events from all services and sends notifications.

**Capabilities:**
- Email sending via SMTP (Gmail)
- SMS sending (via external provider)
- Audit logging of all entity changes
- Notification logging for delivery tracking
- Event-driven notification dispatch

### 3. ProcessNotificationCommand & Handler

**Location:** `NotificationService.Application/Commands/ProcessNotificationCommandHandler.cs`

**What it is:** Generic command to process various notification events.

```csharp
public class ProcessNotificationCommandHandler : IRequestHandler<ProcessNotificationCommand, ApiResponse<bool>>
{
    private readonly IEmailSender _emailSender;
    private readonly INotificationDbContext _context;
    
    public async Task<ApiResponse<bool>> Handle(
        ProcessNotificationCommand request, 
        CancellationToken cancellationToken)
    {
        try
        {
            switch (request.NotificationType)
            {
                case NotificationType.Email:
                    return await SendEmailAsync(request);
                    
                case NotificationType.Sms:
                    return await SendSmsAsync(request);
                    
                default:
                    return ApiResponse<bool>.Fail("Unknown notification type");
            }
        }
        catch (Exception ex)
        {
            // Log the failure
            await LogNotificationAsync(request, false, ex.Message);
            return ApiResponse<bool>.Fail($"Notification failed: {ex.Message}");
        }
    }
    
    private async Task<ApiResponse<bool>> SendEmailAsync(ProcessNotificationCommand request)
    {
        var message = new EmailMessage
        {
            To = request.Recipient,
            Subject = request.Subject,
            Body = request.Body,
            IsHtml = request.IsHtml
        };
        
        var result = await _emailSender.SendAsync(message);
        
        // Log the attempt
        await LogNotificationAsync(request, result.Success, result.ErrorMessage);
        
        if (result.Success)
            return ApiResponse<bool>.Success(true, "Email sent");
            
        return ApiResponse<bool>.Fail(result.ErrorMessage ?? "Failed to send email");
    }
    
    private async Task LogNotificationAsync(
        ProcessNotificationCommand request, 
        bool success, 
        string? error)
    {
        var log = new NotificationLog
        {
            Id = Guid.NewGuid(),
            UserId = request.UserId,
            Recipient = request.Recipient,
            Subject = request.Subject,
            Body = request.Body,
            Type = request.NotificationType.ToString(),
            IsSuccess = success,
            ErrorMessage = error,
            TraceId = request.TraceId,
            CreatedAtUtc = DateTime.UtcNow
        };
        
        _context.NotificationLogs.Add(log);
        await _context.SaveChangesAsync(cancellationToken);
    }
}
```

```csharp
public class NotificationDbContext : DbContext, INotificationDbContext
```

### 2. GmailSmtpEmailSender.cs

**Location:** `NotificationService.Infrastructure/Persistence/EmailSender/GmailSmtpEmailSender.cs`

**What it teaches:** How to send emails via SMTP.

```csharp
public class GmailSmtpEmailSender : IEmailSender
{
    private readonly SmtpSettings _settings;
    private readonly ILogger<GmailSmtpEmailSender> _logger;
    
    public GmailSmtpEmailSender(
        IOptions<SmtpSettings> settings,
        ILogger<GmailSmtpEmailSender> logger)
    {
        _settings = settings.Value;
        _logger = logger;
    }
    
    public async Task<EmailSendResult> SendAsync(EmailMessage message)
    {
        try
        {
            using var client = new SmtpClient
            {
                Host = _settings.Host,
                Port = _settings.Port,
                EnableSsl = true,
                Credentials = new NetworkCredential(
                    _settings.Username, 
                    _settings.Password)
            };
            
            var mailMessage = new MailMessage
            {
                From = new MailAddress(_settings.FromEmail, _settings.FromName),
                Subject = message.Subject,
                Body = message.Body,
                IsBodyHtml = message.IsHtml
            };
            
            mailMessage.To.Add(message.To);
            
            await client.SendMailAsync(mailMessage);
            
            _logger.LogInformation("Email sent to {Recipient}: {Subject}", 
                message.To, message.Subject);
            
            return new EmailSendResult 
            { 
                Success = true,
                MessageId = Guid.NewGuid().ToString()
            };
        }
        catch (Exception ex)
        {
            _logger.LogError(ex, "Failed to send email to {Recipient}", message.To);
            
            return new EmailSendResult
            {
                Success = false,
                ErrorMessage = ex.Message
            };
        }
    }
}

// Configuration class
public class SmtpSettings
{
    public string Host { get; set; } = "smtp.gmail.com";
    public int Port { get; set; } = 587;
    public string Username { get; set; } = "";
    public string Password { get; set; } = "";
    public string FromEmail { get; set; } = "";
    public string FromName { get; set; } = "CredVault";
}
```

```csharp
    Timestamp = DateTime.UtcNow
});
```
