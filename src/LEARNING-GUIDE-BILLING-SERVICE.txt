# BILLING SERVICE - COMPLETE LEARNING GUIDE
**Port: 5003 | Database: credvault_billing**

---

## What This Service Does

The Billing Service handles **bills, statements, and reward programs**. It's the financial core of the system.

**Capabilities:**
- Generate monthly bills from card transactions
- Track bill payment status
- Mark bills as paid/overdue
- Manage reward tiers (earning rates)
- Track user reward accounts
- Process reward transactions (earn, redeem, reverse)
- Generate monthly statements
- Background job for overdue bill checking

### 4. GenerateAdminBillCommand & Handler (Bill Generation)

**Location:** `BillingService.Application/Commands/Bills/GenerateAdminBillCommandHandler.cs`

**What it teaches:**
- How bills are generated from card balances
- Due date calculation
- Minimum payment calculation

```csharp
public class GenerateAdminBillCommandHandler : IRequestHandler<GenerateAdminBillCommand, ApiResponse<BillDto>>
{
    private readonly IBillRepository _billRepository;
    private readonly ICardServiceClient _cardServiceClient;  // HTTP client to Card Service
    
    public async Task<ApiResponse<BillDto>> Handle(GenerateAdminBillCommand request, CancellationToken cancellationToken)
    {
        // STEP 1: Check if active bill already exists for this card
        var existingBill = await _billRepository.GetActiveBillForCardAsync(request.CardId);
        if (existingBill != null)
        {
            return ApiResponse<BillDto>.Fail("Active bill already exists for this card");
        }
        
        // STEP 2: Get card details from Card Service
        var card = await _cardServiceClient.GetCardAsync(request.CardId);
        if (card == null)
        {
            return ApiResponse<BillDto>.Fail("Card not found");
        }
        
        // STEP 3: Check for outstanding balance
        if (card.OutstandingBalance <= 0)
        {
            return ApiResponse<BillDto>.Fail("No outstanding balance to bill");
        }
        
        // STEP 4: Calculate billing period
        var billingDate = DateTime.UtcNow;
        var dueDate = billingDate.AddDays(20);  // Typically 20 days after billing
        
        // STEP 5: Calculate minimum payment
        // Usually min(5% of balance, fixed minimum like ₹200)
        var minDue = Math.Max(card.OutstandingBalance * 0.05m, 200);
        
        // STEP 6: Create bill
        var bill = new Bill
        {
            Id = Guid.NewGuid(),
            UserId = card.UserId,
            CardId = request.CardId,
            CardNetwork = card.Network,
            IssuerId = card.IssuerId,
            Amount = card.OutstandingBalance,
            MinDue = minDue,
            Currency = "INR",
            BillingDateUtc = billingDate,
            DueDateUtc = dueDate,
            Status = BillStatus.Pending,
            CreatedAtUtc = DateTime.UtcNow,
            UpdatedAtUtc = DateTime.UtcNow
        };
        
        await _billRepository.AddAsync(bill);
        
        // STEP 7: Publish event
        await _publishEndpoint.Publish(new IBillGenerated
        {
            BillId = bill.Id,
            UserId = bill.UserId,
            CardId = bill.CardId,
            Amount = bill.Amount,
            DueDate = bill.DueDateUtc,
            Timestamp = DateTime.UtcNow
        });
        
        return ApiResponse<BillDto>.Success(
            BillMapping.ToDto(bill, card),
            "Bill generated successfully"
        );
    }
}
```

### 5. MarkBillPaidCommand & Handler (Payment Recording)

**Location:** `BillingService.Application/Commands/Bills/MarkBillPaidCommandHandler.cs`

**What it teaches:**
- How payments update bills
- How rewards are calculated on payment
- Transaction management with Unit of Work

```csharp
public class MarkBillPaidCommandHandler : IRequestHandler<MarkBillPaidCommand, ApiResponse<BillDto>>
{
    private readonly IBillRepository _billRepository;
    private readonly IRewardRepository _rewardRepository;
    private readonly IUnitOfWork _unitOfWork;
    
    public async Task<ApiResponse<BillDto>> Handle(MarkBillPaidCommand request, CancellationToken cancellationToken)
    {
        // STEP 1: Get bill
        var bill = await _billRepository.GetByIdWithDetailsAsync(request.BillId);
        if (bill == null)
        {
            return ApiResponse<BillDto>.Fail("Bill not found");
        }
        
        // STEP 2: Validate
        if (bill.UserId != request.UserId)
            return ApiResponse<BillDto>.Fail("Unauthorized");
        
        if (bill.Status == BillStatus.Paid)
            return ApiResponse<BillDto>.Fail("Bill already paid");
        
        // STEP 3: Begin transaction
        await _unitOfWork.BeginTransactionAsync();
        
        try
        {
            // STEP 4: Update bill
            bill.AmountPaid = request.Amount;
            bill.PaidAtUtc = DateTime.UtcNow;
            bill.UpdatedAtUtc = DateTime.UtcNow;
            
            // Determine status based on amount
            if (request.Amount >= bill.Amount)
            {
                bill.Status = BillStatus.Paid;
            }
            else if (request.Amount >= bill.MinDue)
            {
                bill.Status = BillStatus.PartiallyPaid;
            }
            else
            {
                // Still pending but with partial payment
                bill.Status = BillStatus.Pending;
            }
            
            await _billRepository.UpdateAsync(bill);
            
            // STEP 5: Calculate and award rewards
            await AwardRewardsAsync(bill, request.Amount);
            
            // STEP 6: Commit transaction
            await _unitOfWork.CommitAsync();
            
            return ApiResponse<BillDto>.Success(
                BillMapping.ToDto(bill),
                "Bill marked as paid"
            );
        }
        catch (Exception ex)
        {
            await _unitOfWork.RollbackAsync();
            return ApiResponse<BillDto>.Fail($"Payment failed: {ex.Message}");
        }
    }
    
    private async Task AwardRewardsAsync(Bill bill, decimal paymentAmount)
    {
        // Get or create reward account
        var account = await _rewardRepository.GetAccountByUserIdAsync(bill.UserId);
        
        if (account == null)
        {
            // Create new reward account
            account = new RewardAccount
            {
                Id = Guid.NewGuid(),
                UserId = bill.UserId,
                PointsBalance = 0,
                CreatedAtUtc = DateTime.UtcNow,
                UpdatedAtUtc = DateTime.UtcNow
            };
            account = await _rewardRepository.CreateAccountAsync(account);
        }
        
        // Get applicable tier
        var tier = await _rewardRepository.GetApplicableTierAsync(bill.CardNetwork, bill.Amount);
        if (tier == null) return;
        
        // Calculate points (e.g., 1 point per ₹100 spent)
        var points = paymentAmount * tier.RewardRate;
        
        if (points > 0)
        {
            // Create reward transaction
            var rewardTransaction = new RewardTransaction
            {
                Id = Guid.NewGuid(),
                RewardAccountId = account.Id,
                BillId = bill.Id,
                Points = points,
                Type = RewardTransactionType.Earned,
                CreatedAtUtc = DateTime.UtcNow
            };
            
            await _rewardRepository.AddTransactionAsync(rewardTransaction);
            
            // Update account balance
            account.PointsBalance += points;
            account.UpdatedAtUtc = DateTime.UtcNow;
            await _rewardRepository.UpdateAccountAsync(account);
        }
    }
}
```

### 7. GenerateStatementCommand & Handler

**Location:** `BillingService.Application/Commands/Statements/GenerateStatementCommandHandler.cs`

**What it teaches:**
- Aggregating transactions into statements
- Calculating opening/closing balances
- Transaction batching

```csharp
public class GenerateStatementCommandHandler : IRequestHandler<GenerateStatementCommand, ApiResponse<StatementDto>>
{
    private readonly IStatementRepository _statementRepository;
    private readonly ICardServiceClient _cardServiceClient;
    private readonly IBillRepository _billRepository;
    
    public async Task<ApiResponse<StatementDto>> Handle(
        GenerateStatementCommand request, 
        CancellationToken cancellationToken)
    {
        // Get card details
        var card = await _cardServiceClient.GetCardAsync(request.CardId);
        
        // Check for existing statement in period
        var existing = await _statementRepository.GetLatestStatementForCardAsync(request.CardId);
        if (existing != null && 
            existing.PeriodStartUtc <= request.PeriodEnd && 
            existing.PeriodEndUtc >= request.PeriodStart)
        {
            return ApiResponse<StatementDto>.Fail("Statement already exists for this period");
        }
        
        // Get transactions for period
        var transactions = await _cardServiceClient.GetTransactionsAsync(
            request.CardId, 
            request.PeriodStart, 
            request.PeriodEnd);
        
        // Get associated bill
        var bill = await _billRepository.GetActiveBillForCardAsync(request.CardId);
        
        // Calculate totals
        var openingBalance = existing?.ClosingBalance ?? 0;
        var totalPurchases = transactions
            .Where(t => t.Type == "Purchase")
            .Sum(t => t.Amount);
        var totalPayments = transactions
            .Where(t => t.Type == "Payment")
            .Sum(t => t.Amount);
        var totalRefunds = transactions
            .Where(t => t.Type == "Refund")
            .Sum(t => t.Amount);
        
        var closingBalance = openingBalance + totalPurchases - totalPayments + totalRefunds;
        
        // Create statement
        var statement = new Statement
        {
            Id = Guid.NewGuid(),
            UserId = card.UserId,
            CardId = request.CardId,
            BillId = bill?.Id,
            StatementPeriod = request.PeriodStart.ToString("MMMM yyyy"),
            PeriodStartUtc = request.PeriodStart,
            PeriodEndUtc = request.PeriodEnd,
            GeneratedAtUtc = DateTime.UtcNow,
            DueDateUtc = bill?.DueDateUtc,
            OpeningBalance = openingBalance,
            TotalPurchases = totalPurchases,
            TotalPayments = totalPayments,
            TotalRefunds = totalRefunds,
            PenaltyCharges = bill?.Status == BillStatus.Overdue ? 300 : 0,  // Late fee
            InterestCharges = 0,  // Calculate based on revolving balance
            ClosingBalance = closingBalance,
            MinimumDue = Math.Max(closingBalance * 0.05m, 200),
            AmountPaid = 0,
            Status = StatementStatus.Generated,
            CardLast4 = card.Last4,
            CardNetwork = card.Network.ToString(),
            IssuerName = card.IssuerName,
            CreditLimit = card.CreditLimit,
            AvailableCredit = card.CreditLimit - closingBalance,
            CreatedAtUtc = DateTime.UtcNow,
            UpdatedAtUtc = DateTime.UtcNow
        };
        
        statement = await _statementRepository.CreateStatementAsync(statement);
        
        // Add transaction details to statement
        var statementTransactions = transactions.Select(t => new StatementTransaction
        {
            Id = Guid.NewGuid(),
            StatementId = statement.Id,
            SourceTransactionId = t.Id,
            Type = t.Type,
            Amount = t.Amount,
            Description = t.Description,
            DateUtc = t.DateUtc,
            CreatedAtUtc = DateTime.UtcNow
        }).ToList();
        
        await _statementRepository.AddTransactionsAsync(statementTransactions);
        
        return ApiResponse<StatementDto>.Success(
            StatementMapping.ToDto(statement, statementTransactions),
            "Statement generated"
        );
    }
}
```

### 2. UnitOfWork Pattern

**Location:** `BillingService.Infrastructure/UnitOfWork/UnitOfWork.cs`

**What it is:** Ensures data consistency across multiple operations.

```csharp
public class UnitOfWork : IUnitOfWork
{
    private readonly BillingDbContext _context;
    private IDbTransaction? _transaction;
    
    public UnitOfWork(BillingDbContext context)
    {
        _context = context;
    }
    
    public async Task BeginTransactionAsync()
    {
        _transaction = await _context.Database.BeginTransactionAsync();
    }
    
    public async Task CommitAsync()
    {
        if (_transaction != null)
        {
            await _transaction.CommitAsync();
            await _transaction.DisposeAsync();
            _transaction = null;
        }
    }
    
    public async Task RollbackAsync()
    {
        if (_transaction != null)
        {
            await _transaction.RollbackAsync();
            await _transaction.DisposeAsync();
            _transaction = null;
        }
    }
    
    public async Task<int> SaveChangesAsync()
    {
        return await _context.SaveChangesAsync();
    }
}
```

#### BillUpdateSagaConsumer.cs

```csharp
public class BillUpdateSagaConsumer : IConsumer<IStartBillUpdate>
{
    private readonly IBillRepository _billRepository;
    private readonly IRewardRepository _rewardRepository;
    private readonly IUnitOfWork _unitOfWork;
    private readonly IPublishEndpoint _publishEndpoint;
    
    public async Task Consume(ConsumeContext<IStartBillUpdate> context)
    {
        var message = context.Message;
        
        var bill = await _billRepository.GetByIdAsync(message.BillId);
        
        if (bill == null)
        {
            await _publishEndpoint.Publish(new IBillUpdateFailed
            {
                CorrelationId = message.CorrelationId,
                BillId = message.BillId,
                Error = "Bill not found"
            });
            return;
        }
        
        await _unitOfWork.BeginTransactionAsync();
        
        try
        {
            // Update bill
            bill.AmountPaid = message.AmountPaid;
            bill.PaidAtUtc = DateTime.UtcNow;
            bill.UpdatedAtUtc = DateTime.UtcNow;
            
            if (message.AmountPaid >= bill.Amount)
                bill.Status = BillStatus.Paid;
            else if (message.AmountPaid >= bill.MinDue)
                bill.Status = BillStatus.PartiallyPaid;
            
            await _billRepository.UpdateAsync(bill);
            
            // Award rewards if applicable
            if (message.RewardPoints > 0)
            {
                var account = await _rewardRepository.GetAccountByUserIdAsync(bill.UserId);
                if (account != null)
                {
                    var transaction = new RewardTransaction
                    {
                        Id = Guid.NewGuid(),
                        RewardAccountId = account.Id,
                        BillId = bill.Id,
                        Points = message.RewardPoints,
                        Type = RewardTransactionType.Earned,
                        CreatedAtUtc = DateTime.UtcNow
                    };
                    await _rewardRepository.AddTransactionAsync(transaction);
                    
                    account.PointsBalance += message.RewardPoints;
                    account.UpdatedAtUtc = DateTime.UtcNow;
                    await _rewardRepository.UpdateAccountAsync(account);
                }
            }
            
            await _unitOfWork.CommitAsync();
            
            // Publish success
            await _publishEndpoint.Publish(new IBillUpdated
            {
                CorrelationId = message.CorrelationId,
                BillId = bill.Id,
                AmountPaid = bill.AmountPaid,
                Status = bill.Status,
                Timestamp = DateTime.UtcNow
            });
        }
        catch (Exception ex)
        {
            await _unitOfWork.RollbackAsync();
            
            await _publishEndpoint.Publish(new IBillUpdateFailed
            {
                CorrelationId = message.CorrelationId,
                BillId = message.BillId,
                Error = ex.Message
            });
        }
    }
}
```

## Background Jobs - Understanding Scheduled Tasks

### OverdueBillScheduler.cs

**Location:** `BillingService.Infrastructure/BackgroundJobs/OverdueBillScheduler.cs`

**What it is:** A background service that checks for overdue bills daily.

```csharp
public class OverdueBillScheduler : BackgroundService
{
    private readonly IServiceProvider _serviceProvider;
    private readonly ILogger<OverdueBillScheduler> _logger;
    
    public OverdueBillScheduler(
        IServiceProvider serviceProvider,
        ILogger<OverdueBillScheduler> logger)
    {
        _serviceProvider = serviceProvider;
        _logger = logger;
    }
    
    protected override async Task ExecuteAsync(CancellationToken stoppingToken)
    {
        _logger.LogInformation("Overdue bill checker started");
        
        while (!stoppingToken.IsCancellationRequested)
        {
            try
            {
                // Run once per day at midnight
                var now = DateTime.UtcNow;
                var nextRun = GetNextRunTime(now);
                var delay = nextRun - now;
                
                if (delay > TimeSpan.Zero)
                {
                    await Task.Delay(delay, stoppingToken);
                }
                
                await CheckOverdueBillsAsync();
            }
            catch (OperationCanceledException)
            {
                break;
            }
            catch (Exception ex)
            {
                _logger.LogError(ex, "Error checking overdue bills");
                await Task.Delay(TimeSpan.FromMinutes(5), stoppingToken);
            }
        }
    }
    
    private async Task CheckOverdueBillsAsync()
    {
        using var scope = _serviceProvider.CreateScope();
        var billRepository = scope.ServiceProvider.GetRequiredService<IBillRepository>();
        
        var now = DateTime.UtcNow;
        var overdueBills = await billRepository.GetOverdueBillsAsync(now);
        
        foreach (var bill in overdueBills)
        {
            if (bill.Status == BillStatus.Pending)
            {
                bill.Status = BillStatus.Overdue;
                bill.UpdatedAtUtc = now;
                await billRepository.UpdateAsync(bill);
                
                // Publish notification event
                _logger.LogInformation("Bill {BillId} marked as overdue", bill.Id);
            }
        }
    }
    
    private DateTime GetNextRunTime(DateTime now)
    {
        // Run at midnight UTC
        return new DateTime(now.Year, now.Month, now.Day, 0, 0, 0, DateTimeKind.Utc)
            .AddDays(1);
    }
}
```

## Exercise: Trace "Pay Bill with Rewards"

1. User initiates payment via Payment Service
2. Payment Service orchestrates saga:
   - OTP verification
   - Bill verification from Billing Service
   - Reward calculation from Billing Service
3. `IStartBillUpdate` event sent to Billing Service
4. `BillUpdateSagaConsumer` receives event
5. Consumer:
   - Updates bill status (Paid/PartiallyPaid)
   - Creates reward transaction (Earned)
   - Updates reward account balance
   - Publishes `IBillUpdated`
6. Saga continues: Card deduction, notification
